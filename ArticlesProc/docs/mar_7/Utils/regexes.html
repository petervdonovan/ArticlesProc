regexes = {
    'long month': r'((January)|(February)|(March)|(April)|(May)|(June)|(July)|(August)|(September)|(October)|(November)|(December))',
    'short month': r'((Jan\.?)|(Feb\.?)|(Mar\.?)|(Apr\.?)|(May\.?)|(June?\.?)|(July?\.?)|(Aug\.?)|(Sept\.?)|(Oct\.?)|(Nov\.?)|(Dec\.?))',
    #'3-letter month no period':r'((Jan)|(Feb)|(Mar)|(Apr)|(May)|(Jun)|(Jul)|(Aug)|(Sep)|(Oct)|(Nov)|(Dec))',
    'season': '((Spring)|(Summer)|(Fall)|(Autumn)|(Winter)|(spring)|(summer)|(fall)|(autumn)|(winter))',
    'recent year': r'20[0-2][0-9]', #in the 2000s
    'citable year': r'(14|15|16|17|18|19|20)[0-9{2}]', #Between 1400 and present
    'name': r'[^\s0-9\.,a-z][^\s0-9\.,?!’\'"”]+',
    'title in quotes': r'["“][^.]*[\.\?!]["”]',
    'title in single quotes': r'[‘\'][^.]*[’\']',
    #'url': r'((http)|(www))', #TODO: DEFINE WHAT MAKES A LINK.
    'url': r'[^ ]*?',
    'day of month': r'([1-3]?[0-9])'
    }
'''Dictionary containing Regex patterns of some common words/phrases.'''
# Names regexes
regexes['surname, given name'] = r'(' + regexes['name'] + r', ' + regexes['name'] + r')'
regexes['surname, given name middle name'] = r'(' + regexes['surname, given name'] + r' ' + regexes['name'] + r')'
regexes['surname, given name and optionally middle name'] = r'(' + regexes['surname, given name'] + r'( ' + regexes['name'] + r')?)'
regexes['surname, given name and optionally middle initial'] = r'(' + regexes['surname, given name'] + r' [^\s0-9\.,a-z]\.' + r')'
regexes['given name optionally middle name surname'] = r'(' + regexes['name'] + r' ' + regexes['name'] + r' (' + regexes['name'] + r')?)'
regexes['given name optionally middle initial surname'] = r'(' + regexes['name'] + r' ([^\s0-9\.,a-z]\. )?' + regexes['name'] + r')'
regexes['surname, first initial'] = r'(' + regexes['name'] + r', [^\b0-9\.,]\.)'
regexes['surname, initials'] = r'(' + regexes['name'] + r' [^\b0-9\.,]{1,2})'
regexes['surname, initials with dot'] = r'(' + regexes['name'] + r',( [^\b0-9\.,]\.){1,2})'
regexes['surname initial(s)'] = r'(' + regexes['name'] + r' [^\b0-9\.,]([^\b0-9\.,]){0,2})'
regexes['surname, initial(s) no dots no spaces'] = r'(' + regexes['name'] + r', [^\s0-9\.,]([^\s0-9\.,]){0,2})'
# Name lists
regexes['ama name list'] = r'(' + regexes['surname initial(s)'] + r'(, ' + regexes['surname initial(s)'] + r')*?((\.)|(, et al.)))'
regexes['apa name list'] = r'(' + regexes['surname, initials'] + r'((, ' + regexes['surname, initials'] + r')*' + \
        r',? & ' + regexes['surname, initials'] + r')?)'
regexes['chicago/turabian name list'] = r'(' + r'(((' + regexes['surname, given name and optionally middle name'] + r')|(' + \
    regexes['surname, given name and optionally middle initial'] + r'))((, (' + regexes['given name optionally middle name surname'] + \
    r'|' + regexes['given name optionally middle initial surname'] + r'))*?, and ' + r'(' + \
    regexes['given name optionally middle name surname'] + r'|' + regexes['given name optionally middle initial surname'] + r')' + r')?\.' + r')' \
    + r'|' + regexes['surname, given name and optionally middle initial'] + r')'
regexes['harvard: australian name list'] = r'(' + regexes['surname, initial(s) no dots no spaces'] + r'((, ' + regexes['surname, initial(s) no dots no spaces'] + r')*' + \
    r' & ' + regexes['surname, initial(s) no dots no spaces'] + r')?)'

regexes['mla name list'] = r'((' + regexes['surname, given name and optionally middle name'] + r'(((\.)|(, et al.))?)|' + \
        regexes['surname, given name and optionally middle initial'] + r'(, et al.)?)' + \
        r'(, (and )?' + regexes['given name optionally middle initial surname'] + r'\.)?' + r')'
regexes['harvard name list'] = r'(' + regexes['surname, initials with dot'] + r'(, ' + \
         regexes['surname, initials with dot'] + r')*( and ' + regexes['surname, initials with dot'] + \
         r')?( et al.)?' + r')'
